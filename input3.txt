int fallspeed(int height, bool hasParachute ){
    float gravity_accel= 9.8;
    float force = height * gravity_accel;
    if(force > 98 && hasParachute == true){
        print("You are safe.");
    }elif(force > 98 && hasParachute == false){
        print("Ouch! You are injured");
    }else {
        print("You didn't need a Parachute!");
    }
}
